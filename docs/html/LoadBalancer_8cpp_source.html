69void LoadBalancer::addRandomRequest() {
70 // Random number generator
71 static std::random_device rd;
72 static std::mt19937 gen(rd());
73 static std::uniform_int_distribution<> chance(1, 100);
74 static std::uniform_int_distribution<> ip_part(1, 254);
75 static std::uniform_int_distribution<> process_time(1, 10);
76 static std::uniform_int_distribution<> burst_chance(1, 100);
77
78 // 30% chance to add new request(s) each tick
79 if (chance(gen) <= 30) {
80 int requestsToAdd = 1;
81
82 // ~1% chance (1 in 100) for burst of 5 requests
83 if (burst_chance(gen) <= 1) {
84 requestsToAdd = 5;
85 std::cout << "Time " << current_time << ": TRAFFIC SURGE! Adding 5 requests" << std::endl;
86 }
87
88 for (int r = 0; r < requestsToAdd; ++r) {
89 // Generate random IP addresses in 192.168.1.xxx
90 std::ostringstream ip_in, ip_out;
91 ip_in << "192.168.1." << ip_part(gen);
92 ip_out << "192.168.1." << ip_part(gen);
93
94 // Create and enqueue new request
95 Request newRequest(ip_in.str(), ip_out.str(), process_time(gen));
96
97 // Check if the request should be blocked
98 if (isBlocked(newRequest.getin())) {
99 logBlockedRequest(newRequest.getin());
100 continue; // Skip adding this request to the queue
101 }
102
103 requestQueue.push(newRequest);
104
105 if (requestsToAdd == 1) {
106 // Normal single request - show details
107 logOutput("Time " + std::to_string(current_time) + ": New request added ("
108 + ip_in.str() + " -> " + ip_out.str()
109 + ", " + std::to_string(newRequest.gettime()) + " cycles)");
110 }
111 }
112
113 if (requestsToAdd > 1) {
114 logOutput("Time " + std::to_string(current_time) + ": Added " + std::to_string(requestsToAdd)
115 + " requests to queue");
116 }
117 }

120void LoadBalancer::tick() {
121 current_time++;
122 logOutput("\n--- Time " + std::to_string(current_time) + " ---");
123
124 // 1. Possibly add a new Request (random chance)
125 addRandomRequest();
126
127 // 2. Manage server load (dynamic scaling)
128 manageServerLoad();
129
130 // 3. For each WebServer:
131 for (size_t i = 0; i < servers.size(); ++i) {
132 WebServer* server = servers[i];
133
134 if (server->isbusy()) {
135 // Get current request info before ticking
136 Request currentRequest = server->getcurr();
137
138 // a. If it's busy: tick down time_remaining
139 server->tick();
140
141 // Check if server just finished
142 if (!server->isbusy()) {
143 logOutput("Server " + std::to_string(i) + ": Completed request ("
144 + currentRequest.getin() + " -> " + currentRequest.getout()
145 + ")!");
146 } else {
147 logOutput("Server " + std::to_string(i) + ": Processing request ("
148 + currentRequest.getin() + " -> " + currentRequest.getout()
149 + "), " + std::to_string(server->gettimeleft()) + " cycles remaining");
150 }
151 } else {
152 // b. If it's free: assign it the next request in the queue
153 if (!requestQueue.empty()) {
154 Request nextRequest = requestQueue.front();
155 requestQueue.pop();
156 server->assignrequest(nextRequest);
157
158 logOutput("Server " + std::to_string(i) + ": Assigned new request ("
159 + nextRequest.getin() + " -> " + nextRequest.getout()
160 + ", " + std::to_string(nextRequest.gettime()) + " cycles)");
161 } else {
162 logOutput("Server " + std::to_string(i) + ": Idle");
163 }
164 }
165 }
166
167 int idle_servers = getIdleServerCount();
168 logOutput("Queue size: " + std::to_string(requestQueue.size())
169 + " | Active servers: " + std::to_string(active_servers) + "/" + std::to_string(max_servers)
170 + " | Idle servers: " + std::to_string(idle_servers));
171}

214void LoadBalancer::scaleDown() {
215 if (active_servers > 1) { // Keep at least 1 server
216 // Find and remove an idle server
217 for (auto it = servers.begin(); it != servers.end(); ++it) {
218 if (!(*it)->isbusy()) {
219 logOutput(">> SCALED DOWN: Removed idle server "
220 + std::to_string(it - servers.begin()));
221 delete *it;
222 servers.erase(it);
223 active_servers--;
224 break;
225 }
226 }
227 }

230void LoadBalancer::manageServerLoad() {
231 int queue_size = requestQueue.size();
232 int idle_servers = getIdleServerCount();
233
234 // Scale up if queue is building up and we have capacity
235 if (queue_size > idle_servers && active_servers < max_servers) {
236 scaleUp();
237 }
238 // Scale down ONLY if queue is empty AND we have many idle servers
239 else if (queue_size == 0 && idle_servers > 2 && active_servers > 1) {
240 scaleDown();
241 }
242}

259void LoadBalancer::loadBlockedIPs(const std::string& filepath) {
260 std::ifstream file(filepath);
261 if (!file.is_open()) {
262 std::cout << "Warning: Could not open blocked IPs file: " << filepath << std::endl;
263 std::cout << "Continuing without IP blocking..." << std::endl;
264 return;
265 }
266
267 std::string ip;
268 int count = 0;
269 while (std::getline(file, ip)) {
270 // Remove any whitespace
271 if (!ip.empty() && ip.back() == '\r') {
272 ip.pop_back(); // Remove carriage return if present
273 }
274
275 if (!ip.empty()) {
276 blockedIPs.insert(ip);
277 count++;
278 }
279 }
280
281 file.close();
282 logOutput("Loaded " + std::to_string(count) + " blocked IP addresses from " + filepath);
283}
